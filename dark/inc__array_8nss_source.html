422 // Sort the array by value according to 'direction' (ASC or DESC).
423 // Supplying a type allows for correct numerical sorting of integers or floats.
424 void Array_Sort(string tag, string dir="ASC", int type=TYPE_STRING, object obj=OBJECT_INVALID)
425 {
426  string table = GetTableName(tag, obj, TRUE);
427  string direction = GetStringUpperCase(dir);
428 
429  if ( ! (direction == "ASC" || direction == "DESC") ) {
430  WriteTimestampedLogEntry("WARNING: Invalid sort direction <" + direction + "> supplied. Defaulting to ASC.");
431  direction = "ASC";
432  }
433 
434  // default orderBy for strings.
435  string orderBy = "ORDER BY value " + direction;
436  switch(type) {
437  case TYPE_INTEGER:
438  orderBy = "ORDER BY CAST(value AS INTEGER)" + direction;
439  break;
440  case TYPE_FLOAT:
441  orderBy = "ORDER BY CAST(value AS DECIMAL)" + direction;
442  break;
443  }
444  ExecuteStatement("CREATE TABLE " +table+ "_temp AS SELECT ROW_NUMBER() OVER(" + orderBy + ")-1, value FROM " +table, obj);
445  ExecuteStatement("DELETE FROM " +table, obj);
446  ExecuteStatement("INSERT INTO " +table+ " SELECT * FROM " +table+ "_temp", obj);
447  ExecuteStatement("DROP TABLE " +table+ "_temp", obj);
448 }
