macro(nwnxlib_scope)
    set(NWNXLIB_SOURCE ${NWNXLIB_SOURCE} PARENT_SCOPE)
endmacro()

macro(nwnxlib_add)
    file(RELATIVE_PATH rel "${CMAKE_SOURCE_DIR}/NWNXLib" "${CMAKE_CURRENT_SOURCE_DIR}")
    foreach(abs ${ARGN})
        if(rel)
            list (APPEND NWNXLIB_SOURCE "${rel}/${abs}")
        else()
            list (APPEND NWNXLIB_SOURCE "${abs}")
        endif()
    endforeach()
    if(rel)
        nwnxlib_scope()
    endif()
endmacro()

nwnxlib_add("Assert.cpp" "Log.cpp" "Serialize.cpp" "Utils.cpp" "Encoding.cpp")
